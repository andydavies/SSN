/***
* Zorgt er voor dat elementen met selector/id obj dezelfde hoogte krijgen
*
*/
function gelijkeHoogte(obj) {
   var maxHeight = 0;
   var breedte = (navigator.appVersion.indexOf("Win") > -1) ? 615 : 631;

   if (document.documentElement.clientWidth > breedte) {
       $(obj).each(function() {
       
          if ($(this).height() > maxHeight) {
             maxHeight = $(this).height();
          }
       });
    		
       $(obj).height(maxHeight);
   }
}
